DROP DATABASE IF EXISTS hotel;
CREATE DATABASE hotel;

DROP TABLE IF EXISTS hotels ;
CREATE TABLE hotels (
    id INT AUTO_INCREMENT, 
    code varchar(20) NOT NULL, 	
    ligging varchar(50),
    gebouwdin SMALLINT,	
    aantalsterren SMALLINT,
    PRIMARY KEY (id)
);
